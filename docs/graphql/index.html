Queries

customer

Description

Retrieve customer address information

Response

Returns a Customer

Arguments
Name Description
customer_id - Int!

Example

Query
query Customer($customer_id: Int!) {
  customer(customer_id: $customer_id) {
    customer_id
    store_id
    first_name
    last_name
    email
    address_id
    active
    address {
      address_id
      address
      address2
      city {
        ...CityFragment
      }
      postal_code
      phone
    }
    payment {
      customer_id
      staff_id
      rental {
        ...RentalFragment
      }
      amount
      payment_date
    }
  }
}

      "address_id": "xyz789",
      "active": true,
      "address": Address,
      "payment": [Payment]
    }
  }
}

Queries

paginatedCustomers

Description

Retrieve a list of customers

Response

Returns a PaginatedCustomers

Arguments
Name Description
offset - Int!
limit - Int!

Example

Query
query PaginatedCustomers(
  $offset: Int!,
  $limit: Int!
) {
  paginatedCustomers(
    offset: $offset,
    limit: $limit
  ) {
    totalCount
    customers {
      customer_id
      store_id
      first_name
      last_name
      email
      address_id
      active
      address {
        ...AddressFragment
      }
      payment {
        ...PaymentFragment
      }
    }
  }
}
Variables
{"offset": 987, "limit": 123}
Response
{
  "data": {
    "paginatedCustomers": {
      "totalCount": 123,
      "customers": [Customer]
    }
  }
}
